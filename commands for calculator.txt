# Install 7-Zip Silently
Write-Host "Installing 7-Zip..."
Start-Process msiexec.exe -ArgumentList "/i C:\Lab\7zip.msi /qn" -Wait

# Verify it exists (Registry Check)
Get-Package -Name "7-Zip*" | Select-Object Name, ProviderName

# Uninstall 7-Zip (We find the IdentifyingNumber/GUID first)
$App = Get-WmiObject Win32_Product | Where-Object { $_.Name -match "7-Zip" }
Write-Host "Uninstalling $($App.Name)..."
Start-Process msiexec.exe -ArgumentList "/x $($App.IdentifyingNumber) /qn" -Wait

# 1. Create the Lab Folder
New-Item -Path "C:\PSLab" -ItemType Directory -Force

# 2. Define the Download URL (Official GitHub Release)
$Url = "https://github.com/PSAppDeployToolkit/PSAppDeployToolkit/releases/download/3.10.0/PSAppDeployToolkit_v3.10.0.zip"
$ZipFile = "C:\PSLab\Toolkit.zip"

# 3. Download the Zip File
Write-Host "Downloading Toolkit..." -ForegroundColor Cyan
[Net.ServicePointManager]::SecurityProtocol = [Net.SecurityProtocolType]::Tls12
Invoke-WebRequest -Uri $Url -OutFile $ZipFile

# 4. Unzip the Files
Write-Host "Extracting Files..." -ForegroundColor Cyan
Expand-Archive -Path $ZipFile -DestinationPath "C:\PSLab" -Force

# 5. Rename the folder for easier access
$ExtractedFolder = Get-ChildItem "C:\PSLab" -Directory | Where-Object { $_.Name -like "PSAppDeployToolkit*" }
Rename-Item -Path $ExtractedFolder.FullName -NewName "Toolkit"

# 6. Clean up
Remove-Item $ZipFile
Write-Host "Installation Complete! Toolkit is located at C:\PSLab\Toolkit" -ForegroundColor Green
